Add new endpoint after GET /files/files/{file_id} endpoint (around line 290):

```python
@router.get("/files/{file_id}/content")
async def get_file_content(
    file_id: str,
    project_id: str = Query(..., description="Project ID for ownership validation"),
    db: AsyncSession = Depends(get_db)
):
    """
    Get raw file content as text.

    Returns the file content as plain text. Only works for text-based file formats.
    For binary files, returns an error.
    """
    try:
        # Import read_file_content from file_service
        from file_service import read_file_content, FileServiceError

        # Read file content (validates ownership internally)
        content_data = await read_file_content(db, file_id, project_id)

        return {
            "file_id": file_id,
            "content": content_data["content"],
            "encoding": content_data.get("encoding", "utf-8")
        }
    except FileServiceError as e:
        raise _handle_file_service_error(e)
    except Exception as e:
        logger.error(f"Unexpected error getting file content: {e}")
        raise HTTPException(status_code=500, detail="Failed to read file content")
```
