  Update frontend3/vite.config.ts to add proxy configuration. The current config defines process.env variables but lacks proxy setup.

  Add proxy.server configuration to forward /api requests to backend:

  ```typescript
  import path from 'path';
  import { defineConfig, loadEnv } from 'vite';
  import react from '@vitejs/plugin-react';

  export default defineConfig(({ mode }) => {
      const env = loadEnv(mode, '.', '');
      return {
        server: {
          port: 3000,
          host: '0.0.0.0',
          proxy: {
            '/api': {
              target: 'http://localhost:8000',
              changeOrigin: true,
              secure: false,
            }
          }
        },
        plugins: [react()],
        define: {
          'process.env.API_KEY': JSON.stringify(env.GEMINI_API_KEY),
          'process.env.GEMINI_API_KEY': JSON.stringify(env.GEMINI_API_KEY)
        },
        resolve: {
          alias: {
            '@': path.resolve(__dirname, '.'),
          }
        }
      };
  });
  ```

  Changes:
  - Added proxy.server object with /api → http://localhost:8000 forwarding
  - changeOrigin: true ensures backend sees requests from localhost:8000
  - secure: false allows self-signed certs (not needed for localhost but good practice)

  This allows frontend code to call fetch('/api/projects') which proxies to http://localhost:8000/api/projects

<task type="auto">
  <name>Task 4: Create API client utility for frontend</name>
  <files>frontend3/lib/api.ts</files>
  <action>
  Create a utility module for API calls (will be used in later phases). This establishes the pattern for backend communication:

  ```bash
  mkdir -p frontend3/lib

  cat > frontend3/lib/api.ts << 'EOF'
  // API client utility for frontend3
  // Base URL for backend API (development)
  const API_BASE = '/api'; // Use proxy in development

  export interface ApiResponse<T> {
    data?: T;
    error?: string;
    status: number;
  }

  // Generic fetch wrapper
  async function apiRequest<T>(
    endpoint: string,
    options: RequestInit = {}
  ): Promise<ApiResponse<T>> {
    try {
      const response = await fetch(`${API_BASE}${endpoint}`, {
        ...options,
        headers: {
          'Content-Type': 'application/json',
          ...options.headers,
        },
      });

      if (!response.ok) {
        return {
          error: `HTTP ${response.status}: ${response.statusText}`,
          status: response.status,
        };
      }

      const data = await response.json();
      return { data, status: response.status };
    } catch (error) {
      return {
        error: error instanceof Error ? error.message : 'Unknown error',
        status: 0,
      };
    }
  }

  // Project APIs
  export const projectApi = {
    list: () => apiRequest('/projects'),
    get: (id: string) => apiRequest(`/projects/${id}`),
    create: (data: unknown) => apiRequest('/projects', {
      method: 'POST',
      body: JSON.stringify(data),
    }),
  };

  // File APIs
  export const fileApi = {
    list: () => apiRequest('/files'),
  };

  // Document APIs
  export const documentApi = {
    list: () => apiRequest('/documents'),
  };

  // Literature APIs
  export const literatureApi = {
    search: (query: string) => apiRequest(`/literature/search?q=${query}`),
  };
  EOF
  ```

  This creates a typed API client that will be used in Phase 11 (View Integration).
