```typescript
export const documentApi = {
  create: async (projectId: string, title: string = 'Untitled Document') =>
    apiRequest('/documents', {
      method: 'POST',
      body: JSON.stringify({ project_id: projectId, title, content: {} }),
    }),

  get: async (documentId: string) =>
    apiRequest(`/documents/${documentId}`),

  update: async (documentId: string, content: any, title?: string) =>
    apiRequest(`/documents/${documentId}`, {
      method: 'PUT',
      body: JSON.stringify({ content, ...(title && { title }) }),
    }),

  delete: async (documentId: string) =>
    apiRequest(`/documents/${documentId}`, {
      method: 'DELETE',
    }),
};
```

**Citation API methods:**
```typescript
export const citationApi = {
  search: async (query: string, limit: number = 10) =>
    apiRequest(`/literature/search?q=${encodeURIComponent(query)}&limit=${limit}`),

  insert: async (documentId: string, paperId: string, position: number) =>
    apiRequest('/citations', {
      method: 'POST',
      body: JSON.stringify({
        document_id: documentId,
        paper_id: paperId,
        position,
      }),
    }),

  generate: async (documentId: string, format: 'apa' | 'mla' | 'chicago' = 'apa') =>
    apiRequest(`/citations/bibliography/${documentId}?format=${format}`),
};
```

**Component structure:**
```typescript
import { useState, useEffect } from 'react';
import { citationApi } from '../lib/api';

interface BibliographyProps {
  documentId: string;
  format?: 'apa' | 'mla' | 'chicago';
}

export const Bibliography: React.FC<BibliographyProps> = ({ documentId, format = 'apa' }) => {
  const [entries, setEntries] = useState<BibliographyEntry[]>([]);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    if (!documentId) return;

    const fetchBibliography = async () => {
      setLoading(true);
      setError(null);

      try {
        const response = await citationApi.generate(documentId, format);
        if (response.error) throw new Error(response.error);

        setEntries(response.data || []);
      } catch (err) {
        console.error('Bibliography error:', err);
        setError(err instanceof Error ? err.message : 'Failed to load bibliography');
      } finally {
        setLoading(false);
      }
    };

    fetchBibliography();
  }, [documentId, format]);

  if (loading) {
    return <div className="text-sm text-slate-600 dark:text-slate-400">Loading bibliography...</div>;
  }

  if (error) {
    return <div className="text-sm text-red-600 dark:text-red-400">{error}</div>;
  }

  if (entries.length === 0) {
    return <div className="text-sm text-slate-500 dark:text-slate-500">No citations in document</div>;
  }

  return (
    <div className="space-y-4">
      <h3 className="text-lg font-bold">References</h3>
      {entries.map((entry) => (
        <div key={entry.id} className="text-sm text-slate-700 dark:text-slate-300 pl-8 -indent-8">
          <span className="font-semibold">{entry.authors}</span>
          ({entry.year}).
          <span className="italic">{entry.title}</span>
          {entry.journal && <>, <em>{entry.journal}</em></>}
          {entry.doi && <>. DOI: {entry.doi}</>}
        </div>
      ))}
    </div>
  );
};
```
