### Task 1: Replace getFileContent() with parseToTipTap() in Workspace.jsx

**File:** `frontend/src/components/layout/Workspace.jsx`

Find the loadDocumentForFile useEffect (lines 194-287). Replace the content loading section (lines 232-255) with proper TipTap parsing.

**Current code to replace (lines 232-255):**
```javascript
} else {
  // Get file content first
  let fileContent = '';
  try {
    const contentResponse = await filesApi.getFileContent(selectedFile.id);
    fileContent = contentResponse.data.content || '';
  } catch (contentError) {
    console.warn('Could not load file content, creating empty document:', contentError);
    fileContent = '';
  }

  // Convert file content to TipTap JSON format
  const paragraphs = fileContent.split('\n\n').filter(p => p.trim());
  const tipTapContent = {
    type: 'doc',
    content: paragraphs.length > 0
      ? paragraphs.map(para => ({
          type: 'paragraph',
          content: [{ type: 'text', text: para.trim() }]
        }))
      : [{ type: 'paragraph', content: [] }]
  };
```

**Replace with:**
```javascript
} else {
  // Parse file content to TipTap JSON format
  let tipTapContent = {
    type: 'doc',
    content: [{ type: 'paragraph', content: [] }] // Default empty document
  };

  try {
    const parseResponse = await filesApi.parseToTipTap(selectedFile.id, selectedProject.id);
    tipTapContent = parseResponse.data.tiptap_json || tipTapContent;
  } catch (parseError) {
    console.warn('Could not parse file content, creating empty document:', parseError);
    // Continue with empty document
  }

  // Create new document for this file with parsed content
```
