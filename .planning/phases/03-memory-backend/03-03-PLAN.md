# Plan 03-03: Memory Query and Retrieval API

**Objective:** Create REST API for accessing and managing claims, findings, preferences, and their relationships

## Context

With the data models (03-01) and service layer (03-02) complete, we need the REST API layer to expose memory functionality to the frontend. This follows the established API pattern in the codebase where each subsystem has its own API router:

- `project_api.py`: Project CRUD and execution
- `file_api.py`: File upload, download, management
- `task_api.py`: Task status and results

The memory API will provide endpoints for:
1. Querying claims and findings
2. Searching memory with filters
3. Retrieving related claims (graph traversal)
4. Managing user preferences

### Existing API Patterns

The codebase uses these patterns (from file_api.py, project_api.py):

**Router structure:**
```python
from fastapi import APIRouter, Depends, HTTPException, Query

    ...

**Response models:**
- Pydantic models for validation
- Exclude unset fields with `exclude_unset=True`
- Nested relationships for related data

**Error handling:**
- `HTTPException(status_code=404, detail="Not found")`
- `HTTPException(status_code=400, detail="Invalid input")`

        offset=offset,

from sqlalchemy import text


@router.get("/projects/{project_id}/claims/{claim_id}/related", response_model=List[ClaimResponse])
async def get_related_claims(
    project_id: str,
    claim_id: str,
    max_depth: int = Query(3, ge=1, le=5, description="Max traversal depth"),
    relationship_type: Optional[str] = Query(None, description="Filter by relationship type"),
    session: AsyncSession = Depends(get_async_session),
):
    """
    Get related claims via graph traversal.

    Uses recursive CTE to traverse claim relationships up to max_depth.
    Returns claims sorted by depth (closest first).
    """
    service = MemoryService(session)

    # Verify claim exists and belongs to project
    claim = await service.get_claim(claim_id)
    if not claim or claim.project_id != project_id:
        raise HTTPException(status_code=404, detail="Claim not found")

    # Call recursive CTE function
    query = text("""
        SELECT * FROM get_related_claims(:project_id, :claim_id, :max_depth)
    """)
    if relationship_type:
        query = text("""
            SELECT * FROM get_related_claims(:project_id, :claim_id, :max_depth)
            WHERE relationship_type = :rel_type
        """)

    result = await session.execute(
        query,
        {"project_id": project_id, "claim_id": claim_id, "max_depth": max_depth, "rel_type": relationship_type}
    )

    related_claim_ids = [row.claim_id for row in result]

    # Fetch full claim objects
    claims = []
    for cid in related_claim_ids:
        c = await service.get_claim(cid)
        if c:
            claims.append(c)

    return [ClaimResponse.model_validate(c) for c in claims]


@router.post("/projects/{project_id}/claims/{from_claim_id}/relationships/{to_claim_id}", response_model=ClaimRelationshipResponse, status_code=201)

    metadata: Optional[Dict[str, Any]] = None,
    session: AsyncSession = Depends(get_async_session),

    """Create a relationship between two claims."""
    service = MemoryService(session)

    # Verify both claims exist
    from_claim = await service.get_claim(from_claim_id)
    to_claim = await service.get_claim(to_claim_id)

    if not from_claim or from_claim.project_id != project_id:
        raise HTTPException(status_code=404, detail="From claim not found")
    if not to_claim or to_claim.project_id != project_id:
        raise HTTPException(status_code=404, detail="To claim not found")

    relationship = await service.create_relationship(
        project_id=project_id,
        from_claim_id=from_claim_id,
        to_claim_id=to_claim_id,
        relationship_type=relationship_type,
        strength=strength,
        metadata=metadata or {},
    )

    return ClaimRelationshipResponse.model_validate(relationship)

    async def search_claims(
        self,
        project_id: str,
        query: str,
        limit: int = 20,
    ) -> List[Claim]:
        """Full-text search on claim_text."""
        from sqlalchemy import or_

        search_pattern = f"%{query}%"
        q = select(Claim).where(
            and_(
                Claim.project_id == project_id,
                or_(
                    Claim.claim_text.ilike(search_pattern),
                    Claim.claim_type.ilike(search_pattern),
                )
            )
        )
        q = q.order_by(Claim.relevance_score.desc().nulls_last())
        q = q.limit(limit)

        result = await self.session.execute(q)
        return list(result.scalars().all())

    async def list_findings(
        self,
        project_id: str,
        analysis_type: Optional[str] = None,
        min_significance: Optional[float] = None,
        limit: int = 100,
    ) -> List[Finding]:
        """List findings with filters."""
        q = select(Finding).where(Finding.project_id == project_id)

        if analysis_type:
            q = q.where(Finding.analysis_type == analysis_type)
        if min_significance is not None:
            q = q.where(Finding.significance >= min_significance)

        q = q.order_by(Finding.significance.desc().nulls_last())
        q = q.limit(limit)

        result = await self.session.execute(q)
        return list(result.scalars().all())

    async def get_preferences(
        self,
        project_id: str,
        category: Optional[str] = None,
    ) -> List[Preference]:
        """Get preferences for project."""
        q = select(Preference).where(Preference.project_id == project_id)

        if category:
            q = q.where(Preference.category == category)

        result = await self.session.execute(q)
        return list(result.scalars().all())

    async def get_preference(
        self,
        project_id: str,
        key: str,
    ) -> Optional[Preference]:
        """Get specific preference."""
        result = await self.session.execute(
            select(Preference).where(
                and_(
                    Preference.project_id == project_id,
                    Preference.key == key
                )
            )
        )
        return result.scalar_one_or_none()

    async def set_preference(
        self,
        project_id: str,
        key: str,
        value: Any,
        category: Optional[str] = None,
    ) -> Preference:

        if pref:
            # Update existing
            pref.value = value
            pref.category = category
            pref.updated_at = datetime.now(timezone.utc)
        else:
            # Create new
            pref = Preference(
                project_id=project_id,
                key=key,
                value=value,
                category=category,
            )
            self.session.add(pref)

        await self.session.flush()
        return pref

    async def delete_preference(
        self,

        key: str,
    ) -> bool:
        """Delete preference."""
        pref = await self.get_preference(project_id, key)
        if not pref:
            return False

        await self.session.delete(pref)
        await self.session.flush()
        return True

            to_claim_id=to_claim_id,
            relationship_type=RelationshipType(relationship_type),

            metadata=metadata or {},
        )
        self.session.add(relationship)
        await self.session.flush()
        return relationship

    async def get_claim_relationships(
        self,
        claim_id: str,
    ) -> List[ClaimRelationship]:
        """Get all relationships for a claim."""
        result = await self.session.execute(
            select(ClaimRelationship).where(
                or_(
                    ClaimRelationship.from_claim_id == claim_id,
                    ClaimRelationship.to_claim_id == claim_id
                )
            )
        )
        return list(result.scalars().all())
```
